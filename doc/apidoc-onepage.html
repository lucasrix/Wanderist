POST /auth
Email registration. A verification email will be sent to the email address provided.

Examples

POST /auth
{
  "email": "username@example.com",
  "password": "password",
  "password_confirmation": "password",
  "confirm_success_url": "http://localhost:3001/",
  "config_name": "default"
}
403
{
  "status": "error",
  "data": {
    "id": null,
    "provider": "email",
    "uid": "",
    "name": null,
    "nickname": null,
    "image": null,
    "email": "username@example.com",
    "created_at": null,
    "updated_at": null
  },
  "errors": {
    "email": [
      "already in use"
    ],
    "full_messages": [
      "Email already in use"
    ]
  }
}

POST /auth/sign_in
Email authentication.

Examples

POST /auth/sign_in
{
  "email": "george@rubygarage.org",
  "password": "password"
}
200
{
  "data": {
    "id": 2,
    "provider": "email",
    "uid": "george@rubygarage.org",
    "name": null,
    "nickname": null,
    "image": null,
    "email": "george@rubygarage.org"
  }
}
POST /auth/sign_in
{
  "email": "george@rubygarage.org",
  "password": "password"
}
201
{
  "data": {
    "id": 2,
    "provider": "email",
    "uid": "george@rubygarage.org",
    "name": null,
    "nickname": null,
    "image": null,
    "email": "george@rubygarage.org"
  }
}
